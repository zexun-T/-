<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <style>
        div {
            height: 40px;
            width: 400px;
            text-align: center;
            line-height: 40px;
            background-color: black;
            color: #fff;
        }
        .text {
            font-size: 20px;
        }
        .cursor {
            display: inline-block;   /*没有设置inline-block，当字体输入时，光标会被挤兑没了*/
            opacity: 1;
            animation: show 1s linear infinite;
        }

        @keyframes show {                 /* 通过设置透明度的变化，实现光标的闪烁*/
            50% {                      
                opacity: 0;
            }
        }
    </style>
</head>

<body>
    <div>
        <span class="text"></span>      <!--创建两个span，一个用来显示文字，一个用来显示光标-->
        <span class='cursor'>_</span>
    </div>
    <script>
        var text = document.getElementsByClassName('text')[0];
        var cursor = document.getElementsByClassName('cursor')[0];
        function print(string) {
            var textArr = Array.from(string);       
            //将输入的字符串输出为数字，当前textArr=['H','e','l','l','o','w','o','r','l','d','!']
            var i = 0;
            len = textArr.length;             //获取textArr的长度，为遍历做准备
            timer = setInterval(function () {
                if (i < len) {
                    cursor.style.animationPlayState = 'paused';    //在输入的时候光标不会闪烁
                    text.innerText += textArr[i];        //将数组内的内容一个一个添加文本上去
                    i++
                }
                if (i >= len) {
                    clearInterval(timer);         //当遍历完后，清除定时器，这个时候，光标继续闪烁
                    cursor.style.animationPlayState = 'running';
                }
            }, 200)
        }
        setTimeout(function(){
            print('Hello,world !')
        },2000)   //先让光标闪两秒，再打印字体上去
    </script>
</body>

</html>

小结：定时器的使用，遍历数组循环。做完后，在此基础上又有了一些稀奇古怪的想法，如果能像打字一样，先上拼音后上字岂不是很神奇，于是动手3小时，又搞了一份。虽然写得真的很不怎么样，而且又很多种方法可以实现，写得很糙，但也确实实现了想要的效果。

基本思路就是有拼音，有字体，那就创建两个数组，一一遍历填上，当第一个拼音数组打印完数组第一个，第一个汉字上场，以此循环，然后判断什么时候打印完即可。但是写的时候就有些卡壳了。慢慢修改成下面的版本。

对我来说，难就难在循环部分，还有重新递归函数，弄到最后，弄了几个循环，慢慢调出来的，收获比较多的自然是递归函数中，递归再次用到的参数给重新定义一下。

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <style>
        div {
            height: 40px;
            width: 400px;
            text-align: center;
            line-height: 40px;
            background-color: black;
            color: #fff;
        }

        .text,
        .change{
            font-size: 20px;
        }

        .cursor {
            display: inline-block;
            opacity: 1;
            animation: show 1s linear infinite;
        }

        @keyframes show {
            50% {
                opacity: 0;
            }
        }
    </style>
</head>

<body>
    <div>
        <span class='change'></span>               <!-- 汉字的部分 -->
        <span class="text"></span>        		<!-- 拼音的部分 -->
        <span class='cursor'>_</span>				<!-- 光标的部分 -->
    </div>
    <script>
        var text = document.getElementsByClassName('text')[0]
        var cursor = document.getElementsByClassName('cursor')[0]
        var change = document.getElementsByClassName('change')[0]
        var pinyin = ['ni', 'hao', 'ya', 'peng', 'you']            //写拼音的数组
        var hanzi =['你','好','呀','朋','友']                   //写汉字的数组
        var pylen = pinyin.length
        var textArr = Array.from(pinyin[0]);       //由于写拼音的数字，里面是字符串，先把第一个字符串变成数组遍历
        var i = 0;
        var j = 1;
        function print(textArr) {
            timer = setInterval(function () {
                cursor.style.animationPlayState = 'paused'
                var len = textArr.length;
                if (i <= len) {                    
                    text.innerHTML += textArr[i];
                    i++;
                    console.log(i)    //写完了拼音，别动，我先给你清除掉
                    if (i > len) {
                        clearInterval(timer);  //清除定时器，你不能循环了
                        text.innerHTML = '' ;   //清空拼音内容
                        change.innerHTML += hanzi[j-1]    //汉字上场
                        console.log(j)                                               
                        cursor.style.animationPlayState = 'running';  
                        if (j < pylen) {     //遍历完第一组拼音，开始进去第二个拼音，于是我一开始设置的j=1
                            i =0       
							//递归的时候，利用到的参数记得初始化，例如这个i用来判断的，前面用了i导致在这里i已经不是从0开始，例如这个textArr,都要初始化一下，重新搞个值给他   
                            textArr = Array.from(pinyin[j]);
                            j++             //当j超过拼音数组的长度，进不来这个判断，于是在上一个判断里面结束罪恶的一生
                            print(textArr)		//递归，现在是第二组拼音，再次处理拼音
                        }
                    }
                }
            }, 200)
        }
        print(textArr)
    </script>
</body>

</html>
