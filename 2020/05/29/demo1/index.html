鼠标点击冒泡特效小功能

功能：初步实现当鼠标左键点击的时候，会产生冒泡效果。

利用原生js点击事件，动态生成DOM节点，通过计时器设置图形冒泡时间。代码如下：

var ohtml = document.getElementsByTagName("html")[0]; //获取节点
var obody = document.getElementsByTagName('body')[0];
//设置点击事件
ohtml.onclick = function(e){
    var x = e.pageX;
    var y = e.pageY;   //获取鼠标的点击位置
    var ele = document.createElement("div"); //每当点击的时候动态生成dom节点
    //设置样式
    ele.style.color = 'red';
    ele.style.zIndex = '999';
    ele.style.position = "absolute"
    ele.style.userSelect = 'none'; //设置不可选中
    ele.style.fontSize = Math.random() * 10 + 25 + "px";  //随机设置文字图形的大小
    //设置图标出现的位置
    ele.style.left = (x-10) + 'px';
    ele.style.top = (y-20) + 'px';
    //将DOM节点插入body中
    obody.appendChild(ele);
    //添加图形进去
    ele.innerText = "♥"
    //设置定时器，计算冒泡消失时间
    var num = 0;
    clearInterval(timer);
    var timer = setInterval(function () {
        num++;
        ele.style.top = y -20 - num + 'px'; //设置冒泡高度
        ele.style.opacity = (150-num)/120; //设置透明度渐变效果
		if (num == 150) {
			clearInterval(timer);  //清空定时器
			obody.removeChild(ele);  //删除节点
        }
    },8);//每8毫秒 num加1，则冒泡出现的时间为1.2秒
}
